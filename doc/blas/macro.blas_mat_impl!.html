macro_rules! blas_mat_impl(
    ($name: ident, $rows: ident, $cols: ident) => (
        impl<T> BlasMatrix<T> for $name<T> {
            #[inline]
            fn order(&self) -> Order { self.order }

            #[inline]
            fn lead_dim(&self) -> uint { self.lead_dim }

            #[inline]
            fn rows(&self) -> uint { self.$rows }

            #[inline]
            fn cols(&self) -> uint { self.$cols }

            #[inline]
            fn as_ptr(&self) -> *const T { self.data.as_ptr() }

            #[inline]
            fn as_mut_ptr(&mut self) -> *mut T { self.data.as_mut_ptr() }
        }

        impl<T> $name<T> {
            #[inline]
            pub fn as_vec(&self) -> &Vec<T> {
                &self.data
            }
        }
    );
)
